fig_solar = go.Figure(
    go.Scattermapbox(
        lat=df_jp_solar['latitude'],
        lon=df_jp_solar['longitude'],
        mode='markers',
        marker=go.scattermapbox.Marker(
            size=df_jp_solar['capacity_mw']/10
        ),
        text=df_jp_solar['name_english'],
    )
)

fig_solar.update_layout(
    mapbox=dict(
        accesstoken=token,
        center=go.layout.mapbox.Center(

)
